# 대입연산자

## 할당 연산자 (=)

```js
const num = 10;
```

## 복합 대입 연산자

```js
let num = 10;
num += 10; // num = num + 10;
console.log(num);
```

```js
let num = 10;
num -= 10; // num = num - 10;
console.log(num);
```

```js
let num = 10;
num *= 10; // num = num * 10;
console.log(num);
```

```js
let num = 10;
num /= 10; // num = num / 10;
console.log(num);
```

```js
let num = 10;
num %= 10; // num = num % 10;
console.log(num);
```
